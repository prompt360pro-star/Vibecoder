### `packages/db/prisma/schema.prisma`

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider  = "postgresql"
  url       = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}

// ═══════════════════════════════════════
// ENUMS
// ═══════════════════════════════════════

enum SubscriptionTier {
  FREE
  PRO
  TEAM
  LIFETIME
}

enum MissionStatus {
  LOCKED
  AVAILABLE
  IN_PROGRESS
  COMPLETED
}

enum IslandLevel {
  BASIC
  INTERMEDIATE
  ADVANCED
  EXPERT
}

enum ViMode {
  TEACHER
  BUILDER
  DETECTIVE
  REVIEWER
  CREATIVE
  QUIZ
  CONVERSATION
  SCANNER
}

enum PostType {
  ACHIEVEMENT
  PROJECT
  PROMPT
  HELP
  GENERAL
}

enum CoopStatus {
  MATCHING
  ACTIVE
  COMPLETED
  ABANDONED
}

enum LearningStyle {
  READING
  WATCHING
  DOING
  TALKING
}

enum MemoryType {
  FACT
  PREFERENCE
  STRUGGLE
  PROJECT
  GOAL
}

// ═══════════════════════════════════════
// USERS & AUTH
// ═══════════════════════════════════════

model User {
  id                   String            @id @default(cuid())
  clerkId              String            @unique @map("clerk_id")
  email                String            @unique
  name                 String
  username             String            @unique
  avatarUrl            String?           @map("avatar_url")
  bio                  String?

  // DNA Profile
  dnaProfile           Json?             @map("dna_profile")
  learningStyle        LearningStyle?    @map("learning_style")
  dailyTimeGoalMinutes Int               @default(30) @map("daily_time_goal_minutes")
  buildGoals           String[]          @default([]) @map("build_goals")

  // Gamification
  currentLevel         Int               @default(1) @map("current_level")
  totalXp              Int               @default(0) @map("total_xp")
  streakDays           Int               @default(0) @map("streak_days")
  streakFreezes        Int               @default(2) @map("streak_freezes")
  longestStreak        Int               @default(0) @map("longest_streak")

  // Subscription
  subscriptionTier     SubscriptionTier  @default(FREE) @map("subscription_tier")
  stripeCustomerId     String?           @unique @map("stripe_customer_id")
  stripeSubscriptionId String?           @unique @map("stripe_subscription_id")

  // Settings
  locale               String            @default("pt-BR")
  timezone             String            @default("America/Sao_Paulo")
  soundEnabled         Boolean           @default(true) @map("sound_enabled")
  hapticsEnabled       Boolean           @default(true) @map("haptics_enabled")
  notifyStreak         Boolean           @default(true) @map("notify_streak")
  notifyNewMission     Boolean           @default(true) @map("notify_new_mission")
  notifySocial         Boolean           @default(true) @map("notify_social")
  notifyNews           Boolean           @default(true) @map("notify_news")

  // Availability
  openToWork           Boolean           @default(false) @map("open_to_work")
  preferredStack       String[]          @default([]) @map("preferred_stack")

  // Timestamps
  createdAt            DateTime          @default(now()) @map("created_at")
  updatedAt            DateTime          @updatedAt @map("updated_at")
  lastActiveAt         DateTime          @default(now()) @map("last_active_at")

  // Relations
  missionProgress      MissionProgress[]
  projects             Project[]
  viConversations      ViConversation[]
  viMemories           ViMemory[]
  achievements         UserAchievement[]
  streakHistory        StreakDay[]
  socialPosts          SocialPost[]
  postLikes            PostLike[]
  postComments         PostComment[]
  sharedPrompts        SharedPrompt[]
  savedPrompts         SavedPrompt[]
  certificates         Certificate[]
  coopSessionsAsA      CoopSession[]     @relation("PlayerA")
  coopSessionsAsB      CoopSession[]     @relation("PlayerB")
  notifications        Notification[]
  pushTokens           PushToken[]

  @@map("users")
}

// ═══════════════════════════════════════
// GAMIFICATION
// ═══════════════════════════════════════

model MissionProgress {
  id                String        @id @default(cuid())
  userId            String        @map("user_id")
  missionId         String        @map("mission_id")
  status            MissionStatus @default(LOCKED)
  score             Int           @default(0)
  xpEarned          Int           @default(0) @map("xp_earned")
  timeSpentSeconds  Int           @default(0) @map("time_spent_seconds")
  attempts          Int           @default(0)
  data              Json?
  completedAt       DateTime?     @map("completed_at")
  createdAt         DateTime      @default(now()) @map("created_at")
  updatedAt         DateTime      @updatedAt @map("updated_at")

  user              User          @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, missionId])
  @@index([userId])
  @@index([missionId])
  @@map("mission_progress")
}

model UserAchievement {
  id              String   @id @default(cuid())
  userId          String   @map("user_id")
  achievementId   String   @map("achievement_id")
  earnedAt        DateTime @default(now()) @map("earned_at")
  data            Json?

  user            User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, achievementId])
  @@index([userId])
  @@map("user_achievements")
}

model StreakDay {
  id                String   @id @default(cuid())
  userId            String   @map("user_id")
  date              DateTime @db.Date
  xpEarned          Int      @default(0) @map("xp_earned")
  missionsCompleted Int      @default(0) @map("missions_completed")
  timeSpentMinutes  Int      @default(0) @map("time_spent_minutes")
  freezeUsed        Boolean  @default(false) @map("freeze_used")

  user              User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, date])
  @@index([userId])
  @@map("streak_days")
}

  @@index([userId, type])
  @@map("vi_memories")
}

model SocialPost {
```

```typescript
${TONE_GUIDELINES}

${SAFETY_GUIDELINES}`;
```

```typescript
export type TaskType =
  | "simple_question"
  | "code_generation"
  | "complex_reasoning"
  | "code_review"
  | "summarization"
  | "quiz_generation"
  | "conversation";

export function selectModel(taskType: TaskType, userTier: string): ModelConfig {
  // Free users get cheaper models
  if (userTier === "FREE") {
    switch (taskType) {
      case "simple_question":
      case "conversation":
        return MODELS["groq-llama"];
      case "code_generation":
      case "quiz_generation":
        return MODELS["gpt-4o-mini"];
      case "complex_reasoning":
      case "code_review":
      case "summarization":
        return MODELS["claude-haiku"];
      default:
        return MODELS["gpt-4o-mini"];
    }
  }

  // Pro/Team users get best models
  switch (taskType) {
    case "simple_question":
    case "conversation":
      return MODELS["groq-llama"];
    case "code_generation":
      return MODELS["gpt-4o"];
    case "complex_reasoning":
    case "code_review":
      return MODELS["claude-sonnet"];
    case "summarization":
    case "quiz_generation":
      return MODELS["claude-haiku"];
    default:
      return MODELS["claude-sonnet"];
  }
}

export function classifyTask(message: string): TaskType {
  const lower = message.toLowerCase();

  if (lower.match(/revise|review|analise|análise|melhore|otimize/)) {
    return "code_review";
  }
  if (lower.match(/crie|gere|faça|construa|implemente|code|código/)) {
    return "code_generation";
  }
  if (lower.match(/por que|como funciona|explique|arquitetura|design|pattern/)) {
    return "complex_reasoning";
  }
  if (lower.match(/quiz|teste|pergunta|avalie/)) {
    return "quiz_generation";
  }
  if (lower.match(/resuma|resumo|sintetize/)) {
    return "summarization";
  }
  if (lower.length < 100) {
    return "simple_question";
  }

  return "conversation";
}
```

```typescript
import { useAuth } from "@clerk/clerk-expo";

const API_URL = process.env.EXPO_PUBLIC_API_URL || "http://localhost:3000/api";

class ApiClient {
  private getToken: (() => Promise<string | null>) | null = null;

  setTokenGetter(fn: () => Promise<string | null>) {
    this.getToken = fn;
  }

  private async getHeaders(): Promise<HeadersInit> {
    const headers: HeadersInit = {
      "Content-Type": "application/json",
    };

    if (this.getToken) {
      const token = await this.getToken();
      if (token) {
        headers["Authorization"] = `Bearer ${token}`;
      }
    }

    return headers;
  }

  async get<T>(path: string): Promise<{ data: T }> {
    const response = await fetch(`${API_URL}${path}`, {
      method: "GET",
      headers: await this.getHeaders(),
    });

    if (!response.ok) {
      throw new ApiError(response.status, await response.text());
    }

    const data = await response.json();
    return { data };
  }

  async post<T>(path: string, body?: unknown): Promise<{ data: T }> {
    const response = await fetch(`${API_URL}${path}`, {
      method: "POST",
      headers: await this.getHeaders(),
      body: body ? JSON.stringify(body) : undefined,
    });

    if (!response.ok) {
      throw new ApiError(response.status, await response.text());
    }

    const data = await response.json();
    return { data };
  }

  async put<T>(path: string, body?: unknown): Promise<{ data: T }> {
    const response = await fetch(`${API_URL}${path}`, {
      method: "PUT",
      headers: await this.getHeaders(),
      body: body ? JSON.stringify(body) : undefined,
    });

    if (!response.ok) {
      throw new ApiError(response.status, await response.text());
    }

    const data = await response.json();
    return { data };
  }

  async delete<T>(path: string): Promise<{ data: T }> {
    const response = await fetch(`${API_URL}${path}`, {
      method: "DELETE",
      headers: await this.getHeaders(),
    });

    if (!response.ok) {
      throw new ApiError(response.status, await response.text());
    }

    const data = await response.json();
    return { data };
  }
}

class ApiError extends Error {
  constructor(
    public status: number,
    public body: string
  ) {
    super(`API Error ${status}: ${body}`);
    this.name = "ApiError";
  }
}

export const api = new ApiClient();

// Hook to initialize API with auth token
export function useApiSetup() {
  const { getToken } = useAuth();

  api.setTokenGetter(getToken);
}
```

```tsx
interface ButtonProps {
```

### `apps/web/app/api/vi/chat/route.ts`

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@clerk/nextjs/server";
import { db } from "@vibecode/db";
import { viMessageSchema } from "@vibecode/shared";
import {
  getViSystemPrompt,
  selectModel,
  classifyTask,
  getAnthropicClient,
  getOpenAIClient,
  getGroqClient,
} from "@vibecode/ai";
import { rateLimit } from "../../../lib/rate-limit";

export async function POST(request: NextRequest) {
  const { userId: clerkId } = await auth();

  if (!clerkId) {
    return NextResponse.json(
      { error: { code: "UNAUTHORIZED", message: "Not authenticated" } },
      { status: 401 }
    );
  }

  // Rate limiting
  const rateLimitResult = await rateLimit(clerkId);
  if (!rateLimitResult.success) {
    return NextResponse.json(
      { error: { code: "RATE_LIMIT", message: "Too many requests" } },
      { status: 429 }
    );
  }

  // Parse and validate body
  const body = await request.json();
  const parsed = viMessageSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json(
      { error: { code: "VALIDATION", message: parsed.error.message } },
      { status: 400 }
    );
  }

  const { content, mode, context, contextId } = parsed.data;

  // Get user info
  const user = await db.user.findUnique({
    where: { clerkId },
    select: {
      id: true,
      name: true,
      currentLevel: true,
      subscriptionTier: true,
      locale: true,
    },
  });

  if (!user) {
    return NextResponse.json(
      { error: { code: "NOT_FOUND", message: "User not found" } },
      { status: 404 }
    );
  }

  // Check free tier message limit
  if (user.subscriptionTier === "FREE") {
    const todayMessages = await db.viConversation.count({
      where: {
        userId: user.id,
        createdAt: { gte: new Date(new Date().setHours(0, 0, 0, 0)) },
      },
    });

    if (todayMessages >= 5) {
      return NextResponse.json(
        {
          error: {
            code: "LIMIT_REACHED",
            message: "Limite de mensagens gratuitas atingido. Upgrade para PRO!",
          },
        },
        { status: 403 }
      );
    }
  }

  // Get user memories for context
  const memories = await db.viMemory.findMany({
    where: { userId: user.id },
    orderBy: { updatedAt: "desc" },
    take: 5,
    select: { content: true },
  });

  // Build system prompt
  const systemPrompt = getViSystemPrompt({
    mode: mode as any,
    userLevel: user.currentLevel,
    userName: user.name,
    locale: user.locale,
    currentMission: context,
    memories: memories.map((m) => m.content),
  });

  // Select model based on task
  const taskType = classifyTask(content);
  const modelConfig = selectModel(taskType, user.subscriptionTier);

  // Get conversation history
  const conversationHistory = body.conversationHistory?.slice(-8) || [];

  try {
    let responseContent: string;
    let tokensUsed = 0;

    if (modelConfig.provider === "anthropic") {
      const client = getAnthropicClient();
      const response = await client.messages.create({
        model: modelConfig.model,
        max_tokens: modelConfig.maxTokens,
        system: systemPrompt,
        messages: [
          ...conversationHistory.map((m: any) => ({
            role: m.role as "user" | "assistant",
            content: m.content,
          })),
          { role: "user", content },
        ],
      });

      responseContent =
        response.content[0].type === "text" ? response.content[0].text : "";
      tokensUsed =
        (response.usage?.input_tokens ?? 0) +
        (response.usage?.output_tokens ?? 0);
    } else {
      // OpenAI or Groq (OpenAI-compatible)
      const client =
        modelConfig.provider === "groq" ? getGroqClient() : getOpenAIClient();

      const response = await client.chat.completions.create({
        model: modelConfig.model,
        max_tokens: modelConfig.maxTokens,
        messages: [
          { role: "system", content: systemPrompt },
          ...conversationHistory.map((m: any) => ({
            role: m.role as "user" | "assistant",
            content: m.content,
          })),
          { role: "user", content },
        ],
      });

      responseContent = response.choices[0]?.message?.content ?? "";
      tokensUsed = response.usage?.total_tokens ?? 0;
    }

    // Save conversation
    await db.viConversation.create({
      data: {
        userId: user.id,
        mode: mode?.toUpperCase() as any ?? "TEACHER",
        context,
        contextId,
        messages: [
          { role: "user", content, timestamp: new Date().toISOString() },
          {
            role: "assistant",
            content: responseContent,
            timestamp: new Date().toISOString(),
          },
        ],
        modelUsed: modelConfig.model,
        tokensUsed,
      },
    });

    return NextResponse.json({
      success: true,
      data: {
        content: responseContent,
        model: modelConfig.id,
        tokensUsed,
      },
    });
  } catch (error: any) {
    console.error("Vi chat error:", error);

    return NextResponse.json(
      {
        error: {
          code: "AI_ERROR",
          message: "Failed to get response from AI",
        },
      },
      { status: 500 }
    );
  }
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@clerk/nextjs/server";
import { db } from "@vibecode/db";
import { completeMissionSchema } from "@vibecode/shared";

export async function POST(
  request: NextRequest,
  { params }: { params: { missionId: string } }
) {
  const { userId: clerkId } = await auth();

  if (!clerkId) {
    return NextResponse.json(
      { error: { code: "UNAUTHORIZED", message: "Not authenticated" } },
      { status: 401 }
    );
  }

  const body = await request.json();
  const parsed = completeMissionSchema.safeParse({
    ...body,
    missionId: params.missionId,
  });

  if (!parsed.success) {
    return NextResponse.json(
      { error: { code: "VALIDATION", message: parsed.error.message } },
      { status: 400 }
    );
  }

  const user = await db.user.findUnique({
    where: { clerkId },
    select: { id: true },
  });

  if (!user) {
    return NextResponse.json(
      { error: { code: "NOT_FOUND", message: "User not found" } },
      { status: 404 }
    );
  }

  const { missionId, score, timeSpentSeconds, data } = parsed.data;

  // Calculate XP based on score
  const baseXp = 50;
  const xpEarned = Math.round(baseXp * (score / 100));

  // Update mission progress
  const progress = await db.missionProgress.upsert({
    where: {
      userId_missionId: { userId: user.id, missionId },
    },
    update: {
      status: "COMPLETED",
      score,
      xpEarned,
      timeSpentSeconds,
      data: data ?? undefined,
      completedAt: new Date(),
      attempts: { increment: 1 },
    },
    create: {
      userId: user.id,
      missionId,
      status: "COMPLETED",
      score,
      xpEarned,
      timeSpentSeconds,
      data: data ?? undefined,
      completedAt: new Date(),
      attempts: 1,
    },
  });

  // Unlock next mission
  // (Logic to determine next mission based on content structure)
  const nextMissionId = getNextMissionId(missionId);
  if (nextMissionId) {
    await db.missionProgress.upsert({
      where: {
        userId_missionId: { userId: user.id, missionId: nextMissionId },
      },
      update: { status: "AVAILABLE" },
      create: {
        userId: user.id,
        missionId: nextMissionId,
        status: "AVAILABLE",
      },
    });
  }

  // Award XP
  await db.user.update({
    where: { id: user.id },
    data: { totalXp: { increment: xpEarned } },
  });

  return NextResponse.json({
    success: true,
    data: {
      progress,
      xpEarned,
      nextMissionId,
    },
  });
}

function getNextMissionId(currentId: string): string | null {
  const num = parseInt(currentId.replace("m", ""), 10);
  if (isNaN(num)) return null;
  const next = num + 1;
  return next <= 160 ? `m${next.toString().padStart(2, "0")}` : null;
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { Webhook } from "svix";
import { db } from "@vibecode/db";

const WEBHOOK_SECRET = process.env.CLERK_WEBHOOK_SECRET!;

export async function POST(request: NextRequest) {
  const headerPayload = Object.fromEntries(request.headers.entries());
  const body = await request.text();

  const wh = new Webhook(WEBHOOK_SECRET);
  let event: any;

  try {
    event = wh.verify(body, {
      "svix-id": headerPayload["svix-id"],
      "svix-timestamp": headerPayload["svix-timestamp"],
      "svix-signature": headerPayload["svix-signature"],
    });
  } catch {
    return NextResponse.json({ error: "Invalid signature" }, { status: 400 });
  }

  const eventType = event.type;

  if (eventType === "user.created") {
    const { id, email_addresses, first_name, last_name, image_url, username } =
      event.data;

    const email = email_addresses[0]?.email_address;
    const name = [first_name, last_name].filter(Boolean).join(" ") || "Vibe Coder";
    const generatedUsername =
      username || email?.split("@")[0] || `user${Date.now()}`;

    await db.user.create({
      data: {
        clerkId: id,
        email,
        name,
        username: generatedUsername,
        avatarUrl: image_url,
      },
    });

    console.log(`✅ User created: ${email}`);
  }

  if (eventType === "user.updated") {
    const { id, email_addresses, first_name, last_name, image_url } =
      event.data;

    await db.user.update({
      where: { clerkId: id },
      data: {
        email: email_addresses[0]?.email_address,
        name: [first_name, last_name].filter(Boolean).join(" "),
        avatarUrl: image_url,
      },
    });
  }

  if (eventType === "user.deleted") {
    const { id } = event.data;

    await db.user.delete({
      where: { clerkId: id },
    });

    console.log(`🗑️ User deleted: ${id}`);
  }

  return NextResponse.json({ success: true });
}
```

```typescript
import { Ratelimit } from "@upstash/ratelimit";
import { Redis } from "@upstash/redis";

const redis = new Redis({
  url: process.env.REDIS_URL!,
  token: process.env.REDIS_TOKEN!,
});

const rateLimiter = new Ratelimit({
  redis,
  limiter: Ratelimit.slidingWindow(30, "1 m"), // 30 requests per minute
  analytics: true,
  prefix: "vibecode:ratelimit",
});

export async function rateLimit(identifier: string) {
  try {
    const result = await rateLimiter.limit(identifier);
    return result;
  } catch {
    // If Redis is down, allow the request
    return { success: true, limit: 30, remaining: 30, reset: 0 };
  }
}
```
